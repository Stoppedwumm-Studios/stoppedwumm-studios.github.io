const express = require('express'); const axios = require('axios'); const app = express(); const port = 3000; // Replace YOUR_ROBLOX_API_URL with the actual API URL const robloxApiUrl = 'https://games.roblox.com/v2/users/1278786562/games?accessFilter=2&limit=10&sortOrder=Asc'; app.get('/', async (req, res) => { try { const response = await axios.get(robloxApiUrl); const games = response.data.data; const buttonsHTML = games.map(game => { return `

${game.name}

${game.description || 'No description available'}

`; }).join(''); const html = ` Roblox Games ${buttonsHTML} `; res.send(html); } catch (error) { console.error('Error fetching Roblox games:', error); res.status(500).send('Internal Server Error'); } }); app.listen(port, () => { console.log(`App listening at http://localhost:${port}`); });
